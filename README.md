## PROGRAM: 
~~~
import numpy as np
n=int(input())
11,12=[],[]
for i in range (n):
    11.append(int(input()))
for i in range (n):
    12.append(int(input()))
value1 = np.append(11)
value2 = np.append(12)
result =  value1*value2
print("Product of two arrays is:",result)
~~~
